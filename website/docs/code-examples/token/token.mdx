### 3. Mint token

<RunnableCode
  language={Language.js}
>
{`const provider = new ethers.Wallet(
  process.env.PRIVATE_KEY,
  ethers.getDefaultProvider(process.env.PROVIDER_NETWORK),
)
const sdk = new ThirdwebSDK(provider)
const token = sdk.getTokenModule(process.env.TOKEN_MODULE_ADDRESS)\n
async function mintToken(amount) {
  try {
    await token.mint(ethers.utils.parseUnits(amount, 18))
    console.log(\`Minted "\${amount}" tokens\`)
  } catch (error) {
    console.error('Failed to mint token. Error:', error)
  }
}\n
mintToken('<AMOUNT_TO_MINT>')`}
</RunnableCode>
